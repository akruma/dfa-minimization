68  enlarge(atoi(tokens[0].c_str()));

72  alphabet.add(tokens[2][0]);
73  enlarge(atoi(tokens[1].c_str()));
74  enlarge(atoi(tokens[0].c_str()));

93  alphabet.add(str[i]);
